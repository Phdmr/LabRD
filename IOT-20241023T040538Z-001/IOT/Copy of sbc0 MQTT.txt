
from gpio import *


def publishTemperature():
	value = (((analogRead(0) - 0) * (100.0 - - 100.00)) / (1023.0 - 0)) + -100.0 
	topic = "Temperature"
	payload= str("%.2f 'C"% (value,))
	qos="1"
	mqttclient.init()
	mqttclient.init()
	mqttclient.publish(topic, payload, qos)

	CLI.exit()

def publishHumidity():
